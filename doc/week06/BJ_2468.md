## 노트

## 정답 코드
- dfs 풀이
```
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

public class Main {

	static int N, cnt, maxH;
	static int[][] map;
	static boolean[][] visited;
	static int MAX = Integer.MIN_VALUE;
	
	static int[] dx = {-1, 1, 0, 0};
	static int[] dy = {0, 0, -1, 1};
	
	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer st;
		
		N = Integer.parseInt(br.readLine());
		map = new int[N][N];
		
		for (int i = 0; i < N; i++) {
			st = new StringTokenizer(br.readLine(), " ");
			for (int j = 0; j < N; j++) {
				map[i][j] = Integer.parseInt(st.nextToken());
				maxH = Math.max(maxH, map[i][j]);
			}
		} // for - insert map
		
		for (int r = 0; r <= maxH; r++) {
			visited = new boolean[N][N];
			cnt = 0;
			
			for (int i = 0; i < N; i++) {
				for (int j = 0; j < N; j++) {
					if (map[i][j] > r && !visited[i][j]) {
						cnt++;
						dfs(i, j, r);
					}
				} //for - j
			} //for - i
			
			MAX = Math.max(MAX, cnt);
		} //for - r
		
		System.out.println(MAX);
	} //main

	private static void dfs(int x, int y, int r) {
		visited[x][y] = true;
		
		for (int d = 0; d < 4; d++) {
			int nx = x + dx[d];
			int ny = y + dy[d];
			
			if (nx >= 0 && ny >= 0 && nx < N && ny < N
					&& !visited[nx][ny] && map[nx][ny] > r) {
				dfs(nx, ny, r);
			}
		} //for - d
	} //dfs
}
```
- bfs 풀이
```
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayDeque;
import java.util.Queue;
import java.util.StringTokenizer;

public class Main {

	static int N, maxH;
	static int[][] map;
	static boolean[][] visited;
	static int MAX = Integer.MIN_VALUE;
	
	static int[] dx = {-1, 1, 0, 0};
	static int[] dy = {0, 0, -1, 1};
	
	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer st;
		
		N = Integer.parseInt(br.readLine());
		map = new int[N][N];
		
		for (int i = 0; i < N; i++) {
			st = new StringTokenizer(br.readLine(), " ");
			for (int j = 0; j < N; j++) {
				map[i][j] = Integer.parseInt(st.nextToken());
				maxH = Math.max(maxH, map[i][j]);
			}
		} // for - insert map
		
		Queue<int[]> que = new ArrayDeque<>();
		
		for (int r = 0; r <= maxH; r++) {
			int cnt = 0;
			visited = new boolean[N][N];
			
			for (int i = 0; i < N; i++) {
				for (int j = 0; j < N; j++) {
					if (map[i][j] > r && !visited[i][j]) {
						cnt++;
						visited[i][j] = true;
						que.offer(new int[] {i, j});
						
						while (!que.isEmpty()) {
							int[] cur = que.poll();
							
							for (int d = 0; d < 4; d++) {
								int nx = cur[0] + dx[d];
								int ny = cur[1] + dy[d];
								
								if (nx >= 0 && ny >= 0 && nx < N && ny < N
										&& !visited[nx][ny] && map[nx][ny] > r) {
									visited[nx][ny] = true;
									que.offer(new int[] {nx, ny});
								}
							} //for - d
						} //while - bfs
					} //if
				} //for - j
			} //for - i
			
			MAX = Math.max(MAX, cnt);
		} //for - r
		
		System.out.println(MAX);
	} //main
}
```

## 회고
- 이 문제는 dfs와 bfs 두 가지 방법으로 풀 수 있었다.
- 그러나 dfs에서 재귀를 사용할 경우 N의 값이 많아지면(100) 스택 오버플로우가 발생하였다.
- 시간 복잡도는 같으나 안정성에서 큐를 사용한 bfs를 사용하는 것이 좋다는걸 알게 되었다.
